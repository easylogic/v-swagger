v-swagger

API Documentation like Swagger

Install

npm install v-swagger

Load

import VueSwagger from 'v-swagger'
Vue.use(VueSwagger)
components: { VueSwagger }

Use

<v-swagger :spec="spec" />
 new Vue({
        el: '#app',
        data: {
          spec: {
            host: "https://www.google.com",
            title: "Google Search",
            description: "Hello, API",
            opened: true,
            request: [{
                method: 'get',    // post, delete or put 
                description: "Google Home",
                url: "/"
              }
            ]
          }
        }
      })
